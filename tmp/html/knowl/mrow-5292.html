\begin{align*} \int f(ax+b) \dee{x} &= \int f(u) \cdot \frac{1}{u'(x)}\dee{u}\\ &= \int \frac{1}{a} f(u) \dee{u}\\ &= \frac{1}{a} \int f(u) \dee{u} & \text{since $a$ is a constant}\\ &= \frac{1}{a} F(u)\bigg|_{u=ax+b} +C & \text{since $F(u)$ is an antiderivative of $f(u)$}\\ &= \frac{1}{a} F(ax+b) +C. \end{align*}
