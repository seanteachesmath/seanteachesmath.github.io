\begin{align*} \diff{}{x}\left\{\frac{x}{\sqrt{x^2+1}} \right\}&= \frac{\sqrt{x^2+1}-x\frac{2x}{2\sqrt{x^2+1}}}{x^2+1}\\ &=\left(\frac{\sqrt{x^2+1}-\frac{x^2}{\sqrt{x^2+1}}}{x^2+1}\right)\cdot\frac{\sqrt{x^2+1}}{\sqrt{x^2+1}}\\ &=\frac{(x^2+1)-x^2}{(x^2+1)^{3/2}}\\ &=\frac{1}{(x^2+1)^{3/2}}=(x^2+1)^{-3/2} \end{align*}
