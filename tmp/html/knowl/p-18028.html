\begin{align*} \int_C \vF\cdot\dee{\vr} &= \int_{\partial T} \big\{\big(\tfrac{3}{2}y^2 + e^{-y} +\sin x\big)\dee{x} +\big(\tfrac{1}{2}x^2+x-x e^{-y}\big)\dee{y}\big\}\\ &= \dblInt_T \big\{ \tfrac{\partial }{\partial x}\big(\tfrac{1}{2}x^2+x-x e^{-y}\big) -\tfrac{\partial }{\partial y}\big(\tfrac{3}{2}y^2 + e^{-y} +\sin x\big) \big\}\dee{x}\dee{y}\\ &= \dblInt_T \big\{ \big(x + 1-e^{-y}\big) - \big(3y - e^{-y} \big) \big\}\dee{x}\dee{y}\\ &= \dblInt_T \big\{ x -3y + 1 \big\}\dee{x}\dee{y} \end{align*}

\begin{align*} \dblInt_T \dee{x}\dee{y} &=\text{Area}(T) = \frac{1}{2}(4)(1) = 2\\ \dblInt_T y\ \dee{x}\dee{y} &= 0 \qquad\text{since $y$ is odd under } y\rightarrow-y\\ \dblInt_T x\ \dee{x}\dee{y} &=\int_0^1\dee{x}\int_{-2x}^{2x} \dee{y}\ x =\int_0^1 4x^2\ \dee{x} =\frac{4}{3} \end{align*}
