\begin{align*} w(s,t)&=(st)^2+(s\cos t)^2+(s\sin t)^2 =s^2(t^2+1)\\ \implies w_s(s,t)&=2s(t^2+1)\qquad\text{and}\qquad w_t(s,t)=s^2(2t) \end{align*}
