\begin{align*} \vnabla\cdot\big(\vnabla(r)\big) &= \pdiff{}{x} \frac{x}{\big(x^2+y^2+z^2\big)^{1/2}} +\pdiff{}{y} \frac{y}{\big(x^2+y^2+z^2\big)^{1/2}}\\ &\hskip1in +\pdiff{}{z} \frac{z}{\big(x^2+y^2+z^2\big)^{1/2}}\\ &= \frac{3}{\big(x^2+y^2+z^2\big)^{1/2}} -\frac{1}{2}\frac{2x^2+2y^2+2z^2}{\big(x^2+y^2+z^2\big)^{3/2}}\\ & =\frac{2}{\big(x^2+y^2+z^2\big)^{1/2}}\\ & =\frac{2}{r} \end{align*}
