\begin{align*} \diff{}{x}\left\{\log\left|x+\sqrt{x^2+a^2}\right|\right\} &=\frac{1}{x+\sqrt{x^2+a^2}}\cdot \left(1+\frac{1}{2\sqrt{x^2+a^2}}\cdot 2x\right)\\ &\hskip2in\mbox{(chain rule)}\\ &=\frac{1+\frac{x}{\sqrt{x^2+a^2}}}{x+\sqrt{x^2+a^2}} =\frac{\frac{\sqrt{x^2+a^2}+x}{\sqrt{x^2+a^2}}}{x+\sqrt{x^2+a^2}}\\ &=\frac{1}{\sqrt{x^2+a^2}}\\ \end{align*}
