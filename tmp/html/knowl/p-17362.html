  1. \(\displaystyle \color{red}{\vnabla\times(\vF+\vG)=\vnabla\times\vF+\vnabla \times\vG} \)
  2. \(\color{red}{\vnabla\times(c\vF)=c\,\vnabla\times\vF} \text{,}\) for any constant \(c\)
  3. \(\displaystyle \color{red}{\vnabla\times(f\vF)=(\vnabla f)\times\vF+f\,\vnabla\times\vF} \)

  4. \(\vnabla\times(\vF\times\vG) =\vF(\vnabla\cdot\vG)-(\vnabla\cdot\vF)\vG +(\vG\cdot\vnabla)\vF-(\vF\cdot\vnabla)\vG\)

    Here

    \begin{equation*} (\vG\cdot\vnabla)\vF =\vG_1\frac{\partial\vF}{\partial x} +\vG_2\frac{\partial\vF}{\partial y} +\vG_3\frac{\partial\vF}{\partial z} \end{equation*}
