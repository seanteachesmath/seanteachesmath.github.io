Solution 15.1.6.4.1.

(a) By definition

\begin{align*} \vnabla\cdot(x\,\hi+y\,\hj+z\,\hk) &=\pdiff{}{x}\big(x\big) +\pdiff{}{y}\big(y\big) +\pdiff{}{z}\big(z\big) = 3\\ \vnabla\times(x\,\hi+y\,\hj+z\,\hk) &=\det\left[\begin{matrix}\hi & \hj & \hk \\ \pdiff{}{x} & \pdiff{}{y} & \pdiff{}{z} \\ x & y & z\end{matrix}\right] = \vZero \end{align*}

(b) By definition

\begin{align*} \vnabla\cdot(xy^2\hi-yz^2\hj+zx^2\hk) &=\pdiff{}{x}\big(xy^2\big) +\pdiff{}{y}\big(-yz^2\big) +\pdiff{}{z}\big(zx^2\big)\\ &=y^2-z^2+x^2\\ \vnabla\times(xy^2\hi-yz^2\hj+zx^2\hk) &=\det\left[\begin{matrix}\hi & \hj & \hk \\ \pdiff{}{x} & \pdiff{}{y} & \pdiff{}{z} \\ xy^2 & -yz^2 & zx^2\end{matrix}\right]\\ &= 2yz\,\hi-2xz\,\hj-2xy\,\hk \end{align*}

(c) By definition

\begin{align*} &\vnabla\cdot\left(\frac{x}{\sqrt{x^2+y^2}}\hi+\frac{y}{\sqrt{x^2+y^2}}\hj\right)\\ &\hskip0.5in=\pdiff{}{x}\left(\frac{x}{\sqrt{x^2+y^2}}\right) +\pdiff{}{y}\left(\frac{y}{\sqrt{x^2+y^2}}\right)\\ &\hskip0.5in= \frac{1}{\sqrt{x^2+y^2}} - \frac{x^2}{{[x^2+y^2]}^{3/2}} +\frac{1}{\sqrt{x^2+y^2}} - \frac{y^2}{{[x^2+y^2]}^{3/2}}\\ &\hskip0.5in=\frac{x^2+y^2\ -x^2\ +x^2+y^2\ -\ y^2}{{[x^2+y^2]}^{3/2}}\\ &\hskip0.5in=\frac{1}{\sqrt{x^2+y^2}}\\ &\vnabla\times\left(\frac{x}{\sqrt{x^2+y^2}}\hi+\frac{y}{\sqrt{x^2+y^2}}\hj\right)\\ &\hskip0.5in=\det\left[\begin{matrix}\hi & \hj & \hk \\ \pdiff{}{x} & \pdiff{}{y} & \pdiff{}{z} \\ \frac{x}{\sqrt{x^2+y^2}} & \frac{y}{\sqrt{x^2+y^2}} & 0\end{matrix}\right]\\ &\hskip0.5in=\left(- \frac{xy}{{[x^2+y^2]}^{3/2}}+ \frac{xy}{{[x^2+y^2]}^{3/2}}\right)\hk\\ &\hskip0.5in= \vZero \end{align*}

(d) By definition

\begin{align*} &\vnabla\cdot\left(-\frac{y}{\sqrt{x^2+y^2}}\hi+\frac{x}{\sqrt{x^2+y^2}}\hj\right)\\ &\hskip0.3in=\pdiff{}{x}\left(-\frac{y}{\sqrt{x^2+y^2}}\right) +\pdiff{}{y}\left(\frac{x}{\sqrt{x^2+y^2}}\right)\\ &\hskip0.3in= \frac{xy}{{[x^2+y^2]}^{3/2}} - \frac{xy}{{[x^2+y^2]}^{3/2}}\\ &\hskip0.3in=0\\ &\vnabla\times\left(-\frac{y}{\sqrt{x^2+y^2}}\hi +\frac{x}{\sqrt{x^2+y^2}}\hj\right)\\ &\hskip0.3in=\det\left[\begin{matrix}\hi & \hj & \hk \\ \pdiff{}{x} & \pdiff{}{y} & \pdiff{}{z} \\ -\frac{y}{\sqrt{x^2+y^2}} & \frac{x}{\sqrt{x^2+y^2}} & 0\end{matrix}\right]\\ &\hskip0.3in=\!\left(\!\frac{1}{\sqrt{x^2+y^2}} - \frac{x^2}{{[x^2+y^2]}^{3/2}} +\frac{1}{\sqrt{x^2+y^2}} - \frac{y^2}{{[x^2+y^2]}^{3/2}}\!\right)\!\hk\\ &\hskip0.3in= \frac{x^2+y^2\ -x^2\ +x^2+y^2\ -\ y^2}{{[x^2+y^2]}^{3/2}}\ \hk\\ &\hskip0.3in= \frac{\hk}{\sqrt{x^2+y^2}} \end{align*}
