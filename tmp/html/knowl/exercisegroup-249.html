\begin{gather*} \bar x=\frac{1}{M}\tripInt_B x\rho(x,y,z)\ \dee{V}\quad \bar y=\frac{1}{M}\tripInt_B y\rho(x,y,z)\ \dee{V}\\ \bar z=\frac{1}{M}\tripInt_B z\rho(x,y,z)\ \dee{V} \end{gather*}
