\begin{align*} \int_{u=0}^{u=2} \frac{u^2}{\sqrt{16-u^2}} \dee{u} &=\int_{\theta=0}^{\theta=\pi/6} \frac{16\sin^2\theta}{\sqrt{16-16\sin^2\theta}} 4\cos\theta\dee{\theta}\\ &=16\int_{0}^{\pi/6} \sin^2\theta\ \dee{\theta}\\ &=8\int_{0}^{\pi/6} \big(1-\cos(2\theta)\big)\ \dee{\theta}\\ &=8 \bigg[\theta-\frac{1}{2}\sin(2\theta)\bigg]_{0}^{\pi/6} =8 \bigg[\frac{\pi}{6}-\frac{1}{2}\cdot \frac{\sqrt{3}}{2} \bigg]\\ & = \frac{4\pi}{3}-2\sqrt{3} \end{align*}
