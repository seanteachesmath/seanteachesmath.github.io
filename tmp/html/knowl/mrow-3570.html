\begin{align*} y(x)&\approx y(2)+y'(2)(x-2)=1+\frac{1}{2}(x-2)\\ \end{align*}

In particular,

\begin{align*} y(2.1)&\approx y(2)+y'(2)(2.1-2)=1+\frac{1}{2}(.1)=\boxed{1.05}\\ \end{align*}

The quadratic approximation to \(y(x)\) at \(x=2\) is

\begin{align*} y(x)&\approx y(2)+y'(2)(x-2)+\frac{1}{2} y''(2)(x-2)^2\\ &=1+\frac{1}{2}(x-2)-\frac{1}{6}(x-2)^2\\ \end{align*}

In particular,

\begin{align*} y(2.1)&\approx y(2)+y'(2)(2.1-2)+\frac{1}{2} y''(2)(2.1-2)^2\\ &=1+\frac{1}{2}(.1)-\frac{1}{6}(.1)^2=\boxed{1.0483} \end{align*}
