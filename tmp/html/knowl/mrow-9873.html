\begin{align*} 0&\leq \sin^2 n \le 1\\ 0&\leq \cos^2 n \le 1\\ \text{So,}\qquad -1&\leq \sin^2 n - \cos^2 n \le 1\\ -\frac12=\left(-1+\frac12\right)&\leq \left(\sin^2 n - \cos^2 n+\frac12\right) \le \left(1+\frac12\right)=\frac32\\ -\frac{1}{2}\cdot \frac{ 1}{2^n}&\leq \frac{\sin^2 n - \cos^2 n+\tfrac12}{2^n} \le \frac{3}{2}\cdot\frac{1}{2^n}\\ 0&\leq \left|\frac{\sin^2 n - \cos^2 n+\tfrac12}{2^n}\right| \le \frac{3}{2^{n+1}} \end{align*}
