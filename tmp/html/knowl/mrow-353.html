\begin{align*} \lim_{x \to \infty} f(x) \pm g(x) &= F \pm G\\ \lim_{x \to \infty} f(x) g(x) &= F G\\ \lim_{x \to \infty} \frac{f(x)}{ g(x) } &= \frac{F}{G} & \text{provided } G \neq 0\\ \end{align*}

and for rational numbers \(r\)

\begin{align*} \lim_{x \to \infty} f(x)^r &= F^r & \text{provided $f(x)^r$ is defined for all $x$} \end{align*}
