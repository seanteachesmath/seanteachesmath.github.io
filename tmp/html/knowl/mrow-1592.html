\begin{align*} \diff{}{x}\left\{\log\big(x+\sqrt{x^2+4}\big)\right\} &=\frac{1}{\textcolor{red}{x+\sqrt{x^2+4}}} \cdot \diff{}{x}\left\{\textcolor{red}{x+\sqrt{x^2+4}}\right\}\\ &=\frac{1}{x+\sqrt{x^2+4}}\cdot\left(1+\frac{2x}{2\sqrt{x^2+4}}\right)\\ &=\frac{1+\frac{x}{\sqrt{x^2+4}}}{x+\sqrt{x^2+4}}\cdot \frac{\sqrt{x^2+4}}{\sqrt{x^2+4}}\\ &=\frac{\sqrt{x^2+4}+x}{x\sqrt{x^2+4}+x^2+4} . \end{align*}
