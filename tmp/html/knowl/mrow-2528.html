\begin{align*} Q(138)&=\frac{C}{2}\\ Ce^{-k\cdot138}&=\frac{C}{2}\\ \left(e^{-k}\right)^{138}&=\frac{1}{2}=2^{-1}\\ e^{-k}&=2^{-\tfrac{1}{138}}\\ \end{align*}

Now, we have a better formula for \(Q(t)\text{:}\)

\begin{align*} Q(t)&=C\left(e^{-k}\right)^t\\ Q(t)&=C\cdot2^{-\tfrac{t}{138}}\\ \end{align*}

Finally, we can evaluate what percentage of the sample decays in a day.

\begin{align*} 100\frac{Q(t)-Q(t+1)}{Q(t)}&=100\frac{C\cdot2^{-\tfrac{t}{138}}-C\cdot2^{-\tfrac{t+1}{138}}}{C\cdot2^{-\tfrac{t}{138}}}\left(\frac{\frac{1}{C}}{\frac{1}{C}}\right)\\ &=100\frac{2^{-\tfrac{t}{138}}-2^{-\tfrac{t+1}{138}}}{2^{-\tfrac{t}{138}}}\\ &=100\left(2^{-\tfrac{t}{138}}-2^{-\tfrac{t+1}{138}}\right)2^{\tfrac{t}{138}}\\ &=100\left(1-2^{-\tfrac{1}{138}}\right)\approx 0.5 \end{align*}
