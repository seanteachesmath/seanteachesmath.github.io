\begin{align*} \sqrt{n^2+5n}-\sqrt{2n^2-5}&=\sqrt{n^2\left(1+\frac5n\right)}-\sqrt{n^2\left(2-\frac{5}{n^2}\right)}\\ &=n\sqrt{1+\frac{5}{n}} - n\sqrt{2-\frac{5}{n^2}}\\ &=n\left(\sqrt{1+\frac{5}{n}} - \sqrt{2-\frac{5}{n^2}}\right)\\ \end{align*}

Now, the limit is clear.

\begin{align*} \lim_{n \to \infty}\left[\sqrt{n^2+5n}-\sqrt{2n^2-5}\right]&= \lim_{n \to \infty}\left[n\left(\sqrt{1+\frac{5}{n}} - \sqrt{2-\frac{5}{n^2}}\right) \right]\\ &=\lim_{n \to \infty}\left[n\left(\sqrt{1+0} - \sqrt{2-0}\right) \right]\\ &=\lim_{n \to \infty}\left[n\left(-1\right) \right]=-\infty \end{align*}
