\begin{align*} \int_2^\infty \frac{1}{x(\log x)^{101}}\dee{x}&=\lim_{b \to \infty}\int_2^b \frac{1}{x(\log x)^{101}}\dee{x}\\ &=\lim_{b \to \infty}\int_{\log 2}^{\log b}\frac{1}{u^{101}}\,\dee{u}\\ &=\lim_{b \to \infty}\left[\frac{-1}{100 u^{100}}\right]_{\log 2}^{\log b}\\\\ &=\frac{1}{100(\log 2)^{100}} \end{align*}
