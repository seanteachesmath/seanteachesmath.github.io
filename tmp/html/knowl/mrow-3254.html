\begin{align*} T_{2n}(x)&=\sum_{k=0}^{2n} \frac{f^{(k)}\left(\frac{\pi}{4}\right)}{k!}\left(x-\frac{\pi}{4}\right)^k\\ &=\sum_{k=0}^{n} \left[ \underbrace{\frac{f^{(2k)}\left(\frac{\pi}{4}\right)}{(2k)!}\left(x-\frac{\pi}{4}\right)^{2k}}_{\mbox{even term}} + \underbrace{\frac{f^{(2k+1)}\left(\frac{\pi}{4}\right)}{(2k+1)!}\left(x-\frac{\pi}{4}\right)^{2k+1}}_{\mbox{odd term}}\right]\\ &=\sum_{k=0}^n \left[\frac{(-1)^k}{(2k)!\sqrt{2}}\left(x-\frac{\pi}{4}\right)^{2k} + \frac{(-1)^k}{(2k+1)!\sqrt{2}}\left(x-\frac{\pi}{4}\right)^{2k+1}\right] \end{align*}
