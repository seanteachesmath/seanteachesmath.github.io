\begin{align*} A&=\int_0^1\arctan x\dee{x} =\underbrace{ x\arctan x}_{uv}\Big|_0^1-\int_0^1\underbrace{\frac{x}{1+x^2}\,\dee{x}}_{v\dee{u}}\\ &=\arctan 1-\half\log(1+x^2)\Big|_0^1\\ &=\frac{\pi}{4}-\frac{\log 2}{2} \end{align*}
