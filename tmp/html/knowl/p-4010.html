\begin{equation*} \begin{array}{c|c|c|c} n & \dfrac{(n+1)!}{1000\cdot 13^{n+1}}&\left|f^{(n+1)}(c)\right| &\dfrac{ (n+1)!}{1000\cdot13^{n+1}} \\ \;\;&\;\;&\;\;\\ \hline \;\;&\;\;&\;\;\\ 0&\dfrac{1}{1000\cdot 13}&|f'(c)|=\dfrac{1}{7c^{6/7}} \lt \dfrac{1}{7\cdot 3^6} & \dfrac{1}{1000\cdot13^1} \\ \;\;&\;\;&\;\;\\ \hline \;\;&\;\;&\;\;\\ 1&\dfrac{2}{1000\cdot 13^2}&|f''(c)|=\dfrac{6}{7^2\cdot c^{\tfrac{13}{7}}} \lt \dfrac{6}{7^2\cdot 3^{13}}& \dfrac{ (2)!}{1000\cdot13^{2}} \end{array} \end{equation*}

Since

\begin{equation*} \frac{7^2\cdot 3^{13}}{3}= 26,040,609>169,000=1000\cdot 13^2 \end{equation*}

we have \(\left|f^{(n+1)}(c)\right| \lt \frac{ (n+1)!}{1000\cdot13^{n+1}}\) when \(n=1\text{.}\)
