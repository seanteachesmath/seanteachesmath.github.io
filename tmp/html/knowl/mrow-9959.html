\begin{align*} \frac{1}{5-x} =\frac{1}{2-(x-3)} &= \frac{1}{2}\ \frac{1}{1-\frac{x-3}{2}} \qquad \text{set $\frac{x-3}{2}=y$}\\ &= \frac{1}{2}\ \frac{1}{1-y}\bigg|_{y=\frac{x-3}{2}} = \frac{1}{2}\ \bigg[\sum_{n=0}^\infty y^n \bigg]_{y=\frac{x-3}{2}} \quad\text{if $|y| \lt 1$}\\ &= \frac{1}{2}\ \sum_{n=0}^\infty {\Big(\frac{x-3}{2}\Big)}^n = \sum_{n=0}^\infty \frac{(x-3)^n}{2^{n+1}}\\ &= \frac{x-3}{2} + \frac{(x-3)^2}{4} + \frac{(x-3)^3}{8} +\cdots \end{align*}
