\begin{align*} \frac{e^{-\sqrt{x}}}{\sqrt{x}} &\geq \frac{e^{\sqrt{n}}}{\sqrt{n}}&\text{for }x\text{ in the interval } [n-1,n]\\ \mbox{So,}\qquad\color{blue}{ \int_{n-1}^n \frac{e^{-\sqrt{x}}}{\sqrt{x}}\ \dee{x}} &\geq\int_{n-1}^n \frac{e^{-\sqrt{n}}}{\sqrt{n}}\ \dee{x}\\ &=\left[\frac{e^{-\sqrt{n}}}{\sqrt{n}}x\right]_{x=n-1}^{x=n}\\ &=\color{red}{ \frac{e^{-\sqrt{n}}}{\sqrt{n}} } \end{align*}
