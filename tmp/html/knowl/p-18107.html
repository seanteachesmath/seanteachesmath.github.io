\begin{equation*} \dee{s} =\sqrt{\dee{x}^2+\dee{y}^2} =\sqrt{1+\Big(\diff{y}{x}\Big)^2}\,\dee{x} =\sqrt{1+4x^2}\,\dee{x} \end{equation*}

\begin{equation*} \int_{\cC} x\,\dee{s}=\frac{1}{2}+\frac{1}{12}\big[5^{3/2}-1\big] \approx 1.3484 \end{equation*}
