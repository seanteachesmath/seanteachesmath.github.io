\begin{align*} \frac{1}{1-x}&=\sum_{n=0}^\infty x^n\\ \diff{}{x}\left\{\frac{1}{1-x} \right\}&=\diff{}{x}\left\{ \sum_{n=0}^\infty x^n \right\}\\ \frac{1}{(1-x)^2}&=\sum_{n=0}^\infty nx^{n-1}=\sum_{n=1}^\infty nx^{n-1}\\ \diff{}{x}\left\{\frac{1}{(1-x)^2} \right\}&=\diff{}{x}\left\{ \sum_{n=1}^\infty nx^{n-1} \right\}\\ \frac{2}{(1-x)^3}&=\sum_{n=1}^\infty n(n-1)x^{n-2}=\sum_{n=2}^\infty n(n-1)x^{n-2}\\ &=\sum_{n=0}^\infty (n+2)(n+1)x^n \end{align*}
