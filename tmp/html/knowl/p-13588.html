\begin{equation*} f_x=3x^2+3y-6=0 \qquad f_y=3x+6y-3 = 0 \end{equation*}

\begin{equation*} 6x^2-3x-9=0 \iff 3(2x-3)(x+1)=0 \iff x=\frac{3}{2},\ -1 \end{equation*}
