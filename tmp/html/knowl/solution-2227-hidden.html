Let \(G(x,y,z) = x^2+y^2+2z\text{.}\) Then, by (14.3.3),

\begin{align*} \hn\,\dee{S}&=\frac{\vnabla G}{\vnabla G\cdot\hk}\ \dee{x}\dee{y} =\frac{2x\hi+2y\hj+2\hk}{2}\,\dee{x}\dee{y} =(x\hi+y\hj+\hk)\,\dee{x}\dee{y} \end{align*}

so that

\begin{align*} \frac{x^2+y^2}{\sqrt{1+x^2+y^2}}\ \dee{S} &=\frac{x^2+y^2}{\sqrt{1+x^2+y^2}}\ \sqrt{1+x^2+y^2}\,\dee{x}\dee{y} =(x^2+y^2)\ \dee{x}\dee{y} \end{align*}

and

\begin{align*} \vF\cdot\hn\,\dee{S} &=\big[x\hi+y\hj+z\hk\big]\cdot\big[x\hi+y\hj+\hk\big]\,\dee{x}\dee{y} =\big[x^2+y^2+z\big]\,\dee{x}\dee{y}\\ &=\Big[1+\frac{1}{2}(x^2+y^2)\Big]\,\dee{x}\dee{y} \end{align*}

since \(z=1-\half(x^2+y^2)\) on \(\cS\text{.}\)

(a)

\begin{align*} \dblInt_\cS \frac{x^2+y^2}{\sqrt{1+x^2+y^2}}\ \dee{S} &=\dblInt_\cS (x^2+y^2)\ \dee{x}\dee{y} =4\int_0^1\dee{x}\int_0^1\dee{y}\ (x^2+y^2)\\ &=4\int_0^1\dee{x}\ \big(x^2+\frac{1}{3}\big) =4\Big(\frac{1}{3}+\frac{1}{3}\Big) =\frac{8}{3} \end{align*}

(b)

\begin{equation*} \dblInt_\cS \vF\cdot\hn\,\dee{S} =\int_{-1}^1\dee{x}\int_{-1}^1\dee{y}\ \Big[1+\frac{1}{2}(x^2+y^2)\Big] =2\times 2+\frac{1}{2}\times\frac{8}{3} =\frac{16}{3} \end{equation*}
