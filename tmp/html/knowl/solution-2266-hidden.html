(a) Since

\begin{align*} &\vnabla\left(\frac{1}{r}\right) = \left(\hi\pdiff{}{x} +\hj\pdiff{}{y} +\hk\pdiff{}{z} \right) \big(x^2+y^2+z^2\big)^{-1/2}\\ &\hskip0.25in=-\hi\frac{x}{\big(x^2+y^2+z^2\big)^{3/2}} -\hj\frac{y}{\big(x^2+y^2+z^2\big)^{3/2}} -\hk\frac{z}{\big(x^2+y^2+z^2\big)^{3/2}}\\ &\hskip0.25in=-\hi\frac{x}{r^3} -\hj\frac{y}{r^3} -\hk\frac{x}{r^3} \end{align*}

we have \(a=-3\text{.}\)

(b) Since

\begin{align*} \vnabla\cdot\big(r\,\vr\big) &= \pdiff{}{x} \Big[\big(x^2+y^2+z^2\big)^{1/2}x\Big] +\pdiff{}{y}\Big[\big(x^2+y^2+z^2\big)^{1/2}y\Big]\\ &\hskip1in +\pdiff{}{z} \Big[\big(x^2+y^2+z^2\big)^{1/2}z\Big]\\ &= 3\big(x^2+y^2+z^2\big)^{1/2} +\frac{1}{2}\frac{2x^2+2y^2+2z^2}{\big(x^2+y^2+z^2\big)^{1/2}}\\ & =4\big(x^2+y^2+z^2\big)^{1/2}\\ & =4r \end{align*}

we have \(a=4\text{.}\)

(c) Since

\begin{align*} &\vnabla(r^3) = \left(\hi\pdiff{}{x} +\hj\pdiff{}{y} +\hk\pdiff{}{z} \right) \big(x^2+y^2+z^2\big)^{3/2}\\ &\hskip0.1in=\hi\,3x\big(x^2+y^2+z^2\big)^{1/2} +\hj\,3y\big(x^2+y^2+z^2\big)^{1/2} +\hk\,3z\big(x^2+y^2+z^2\big)^{1/2}\\ &\hskip0.1in=3r\vr \end{align*}

we have

\begin{align*} \vnabla\cdot\big(\vnabla(r^3)\big) &= \vnabla\cdot\big(3r\vr\big) = 3\,\vnabla\cdot\big(r\vr\big) = 3\,\big(4r\big) \qquad\text{by part (b)}\\ &= 12 r \end{align*}

so that \(a=12\text{.}\)
