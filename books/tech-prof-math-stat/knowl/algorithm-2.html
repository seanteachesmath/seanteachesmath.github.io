Algorithm 3.3.2. R code for calculating Fibonacci numbers. 
x0 <- 1
x1 <- 1
xs <- c(x0, x1)
for(i in 1:n){
  x2 <- x0 + x1     ## calculate the sum
  xs <- c(xs, x2)   ## store the new number
  x0 <- x1          ## replace the oldest number by the second oldest
  x1 <- x2          ## replace the second oldest by the newest, repeat
}
